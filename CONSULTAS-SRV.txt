1.- Mostrar las ganancias de dinero obtenidos en algun mes de las rentas que ya esten finalizadas (conste que lo unico que importa es la fecha de retiro en esta ocasion)

select datediff(r.fecha_devolucion,r.fecha_retiro) as dias from Renta as r; //diferencia de dias en renta

(select r.rut_cliente,(datediff(r.fecha_devolucion,r.fecha_retiro)*v.precio) as precio from Renta as r inner join Vehiculo as v on r.matricula=v.matricula);  //tengo el precio total y rut del cliente

---mysql>  select r.rut_cliente,(datediff(r.fecha_devolucion,r.fecha_retiro)*v.precio) as precio from Renta as r inner join Vehiculo as v on r.matricula=v.matricula where r.estado='Finalizada' and r.fecha_retiro between ('2019-01-01') and ('2019-05-02');
