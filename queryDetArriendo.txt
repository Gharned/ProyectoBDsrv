query para conseguir detalles del arriendo

---esta es para conceguir calle,numero,ciudad,region de la sucursal retiro

select calle,numero,ciudad,region from Sucursal as s inner join Direccion_sucursal as ds on s.id_sucursal=ds.id_sucursal;
